### node后端进行判断
#### 路由判断
```
app.use(function (req, res ,next) {
  if (req.cookies.userId) {
    next();
  } else {
    console.log(`path:${req.path},orginaUrl:${req.originalUrl}`);
    if (req.originalUrl === '/users/login' || req.originalUrl === '/users/logout' || req.path === "/goods/list" ) { //req.originalUrl.indexOf("/goods/list")>-1
      next();
    } else {
      res.json({

        result: ''
      });
    }
  }
});
```
#### 登录接口代码：
```
// 登录接口
router.post("/login", function (req, res, next) {
  let param = {
    userName: req.body.userName,
    userPwd: req.body.userPwd
  }
  User.findOne(param, function (err, doc) {
    if (err) {
      res.json({
        status: '1',
        msg: err.message
      });
    } else {
      if (doc) {
        res.cookie("userId", doc.userId, {
          path: '/',
          maxAge: 1000*60*60
        });
        res.cookie("userName", doc.userName, {
          path: '/',
          maxAge: 1000*60*60
        });
        // req.session.user = doc;
        res.json({
          status: '0',
          msg: '',
          result: {
            userName: doc.userName,
          }
        })
      }
    }
  })
});
```
#### 登出接口代码：
```
// 登出接口
router.post("/logout", function (req, res, next) {
  res.cookie("userId", "", {
    path: "/",
    maxAge: -1
  });
  res.json({
    status: "0",
    msg: "",
    result: ""
  })
});
```

router.beforeEach((to, from, next) => {
  // 判断该路由是否需要登录权限
  if (to.meta.requireAuth) {
    // 通过vuex state获取当前的token是否存在
    if (sessionStorage.getItem('login') === '100') {
      next();
    } else {
      next({
        path: '/login',
        query: {redirect: to.fullPath}  // 将跳转的路由path作为参数，登录成功后跳转到该路由
      });
    }
  } else {
    next();
  }
